def will_block= (wants_blocking)
  require "fcntl"
  flags = fcntl(Fcntl::F_GETFL, 0)
  if wants_blocking
    flags &= ~Fcntl::O_NONBLOCK
  else
    flags |= Fcntl::O_NONBLOCK
  end
  fcntl(Fcntl::F_SETFL, flags)
end
